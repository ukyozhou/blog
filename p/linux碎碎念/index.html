#找到并打开如下文件 ：
 sudo nano /etc/default/grub

#添加或者修改下面这行，并保存退出
 GRUB_DISABLE_OS_PROBER=false

#运行：
 sudo os-prober

#正常会出现如下的输出：
 /dev/nvme0n1p3 : Windows11 : Windows : chain

#重新生成grub配置并重新启动
 sudo update-grub
